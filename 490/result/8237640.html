bool ok (int n, int m, pair<int, int> t) {
   return (t.first >= 1 && t.first <= n) && (t.second >= 1 && t.first <= m);
}
 
const int inf = 1e10 + 7;
void solve() {
   int n, m;
   cin >> n >> m;
   vector<pair<int, int>> res(n), free(m);
   
   map<pair<int, int>, int> res2;
   
   for (int i = 0; i < n; ++i) {
      cin >> res[i].first >> res[i].second;
      res2[res[i]] = 1;
   }
 
   for (int i = 0; i < m; ++i) {
      cin >> free[i].first >> free[i].second;
   }
 
   
   if (n <= 1000 && m <= 1000) {
      int ans = -inf;
      for (int i = 0; i < m; ++i) {
         int cur = inf;
         for (int j = 0; j < n; ++j) {
            cur = min(cur, abs(res[j].first - free[i].first) + abs(res[j].second - free[i].second));
         }
         ans = max(ans, cur);
      }
      cout << ans << '\n';
      return;
   }
 
   map<pair<int, int>, int> dist;
   queue<pair<int, int>> q;
 
   for (int i = 0; i < m; ++i) {
      q.push(free[i]);
      dist[free[i]] = 0;
   }
 
   vector<pair<int, int>> mv = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
   
   int ans = 0;
   while (!q.empty()) {
      pair<int, int> v = q.front();
      q.pop();
 
      if (res2[v]) {
         ans = max(ans, dist[v]);
         continue;
      }
 
      if (dist[v] >= 5) {
         continue;
      }
 
      for (auto k : mv) {
         pair<int, int> nw = {v.first + k.first, v.second + k.second};
         if (!dist.count(nw)) {
            dist[nw] = dist[v] + 1;
            q.push(nw);
         }
      }
   }
 
   cout << ans << '\n';
}
 
signed main() {
   init_code();
   solve();
   return 0;
}
