#include <bits/stdc++.h>

#define fi first
#define se second

using namespace std;
using ll = long long;
using pic = pair<int, char>;

void solve() {
    string s;
    cin >> s;

    int n = s.size();
    int f[128] = {};
    for (char c : s) ++f[c];

    set<pic> st;
    for (char c = 'A'; c <= 'Z'; ++c) if (f[c] != 0) st.insert({f[c], c});
    
    auto remove = [&](char c) {
        st.erase({f[c], c});
        if (--f[c] > 0) st.insert({f[c], c});
    };

    vector<pair<char, char>> ans;
    while (st.size() >= 2) {
        auto it = st.end();
        --it;
        auto it2 = it;
        --it2;
        char c = it->se;
        char d = it2->se;
        ans.push_back({c, d});
        remove(c);
        remove(d);
    }

    cout << ans.size() << '\n';
    for (auto &s : ans) cout << s.fi << s.se << '\n';
}

int main() {
    cin.tie(0) -> sync_with_stdio(0);

    solve();

    return 0;
}
