#include <bits/stdc++.h>

using namespace std;
using ll = int64_t;

struct Data {
    vector<ll> adj;
    bool done[200];
};
vector<Data> liz;

void reset() {
    for(auto& e : liz) {
        fill(e.done, e.done + 200, false);
    }
}

// optimal solution is 100% a graph matrix
// but like no?

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    ll n, m, q;
    cin >> n >> m >> q;
    liz.resize(n + 1);
    for(ll i = 0; i < m; i++) {
        ll a, b;
        cin >> a >> b;
        liz[a].adj.push_back(b);
        liz[b].adj.push_back(a);
    }

    for(ll i = 0; i < q; i++) {
        ll a, b, t;
        cin >> a >> b >> t;

        // lol
        priority_queue<pair<ll, ll>> q;
        q.emplace(0, a);

        bool found = false;
        while(!q.empty()) {
            auto f = q.top();
            q.pop();

            if(f.first == t && f.second == b) {
                cout << "YES\n";
                found = true;
                break;
            }

            if(f.first != t) {
                for(auto& e : liz[f.second].adj) {
                    if(!liz[e].done[f.first + 1]) {
                        q.emplace(f.first + 1, e);
                        liz[e].done[f.first + 1] = true;
                    }
                }
            }
        }
        if(!found)
            cout << "NO\n";

        reset();
    }

    return 0;
}
