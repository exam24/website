#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;
using ll = long long;
int t[200010];
vector<pair<ll,ll>> v[200010];
int z[200010];
priority_queue<pair<ll,ll>> q;

int main() {
  ll n, a, b, c;
  cin >> n;
  for (int i = 1; i <= n; ++i) {
    cin >> t[i];
    if (t[i]) z[i] = 1000000007;
  }
  for (int i = 0; i < n-1; ++i) {
    cin >> a >> b >> c;
    v[a].push_back({b,c});
    v[b].push_back({a,c});
    if (t[a] && !t[b]) {
      q.push({-c, a});
    }
    else if (!t[a] && t[b]) {
      q.push({-c, b});
    }
  }
  ll s = 0;
  while (q.size()) {
    pair<ll,ll> p = q.top();
    //cout << p.first << " " << p.second << endl;
    q.pop();
    if (-p.first > z[p.second]) continue;
    z[p.second] = -p.first;
    for (auto u : v[p.second]) {
      if (!t[u.first]) continue;
      q.push({-u.second + p.first, u.first});
    }
  }
  for (int i = 1; i <= n; ++i) {
    //cout << i << " " << z[i] << endl;
    s += z[i];
  }
  cout << s << endl;
}
