typedef long long ll;

#define M 1000000007
#define N (1 << 18)

bool t[200001];
vector<pair<int, ll>> v[200001];

ll dst[200001];
bool vis[200001];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> t[i];
        dst[i] = 1e9;
    }

    for (int i = 1; i < n; ++i) {
        int a, b, c;
        cin >> a >> b >> c;
        v[a].push_back({b, c});
        v[b].push_back({a, c});
    }

    priority_queue<pair<ll, int>> q;

    for (int i = 1; i <= n; ++i) {
        if (!t[i]) {
            q.push({0, i});
            dst[i] = 0;
        }
    }

    while (!q.empty()) {
        auto [d, s] = q.top(); q.pop();

        if (vis[s]) continue;
        vis[s] = 1;

        for (auto [u, c] : v[s]) {
            if (dst[u] > dst[s] + c) {
                dst[u] = dst[s] + c;
                q.push({-dst[u], u});
            }
        }
    }

    ll ans = 0;

    for (int i = 1; i <= n; ++i) ans += dst[i];

    cout << ans << "\n";

    return 0;
}
