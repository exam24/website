#include <bits/stdc++.h>

typedef long long ll;

#define M 1000000007
#define N (1 << 18)

using namespace std;

string mp[101];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int n, m;
    cin >> n >> m;
    pair<int, int> tl, br;
    for (int i = 0; i < n; ++i) {
        cin >> mp[i];
    }

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (mp[i][j] == '*' && mp[i][j + 1] == '*' && mp[i + 1][j] == '*') tl = {i, j};
            if (mp[i][j] == '*' && mp[i][j - 1] == '*' && mp[i - 1][j] == '*') br = {i, j};
        }
    }

    int ans = 0;

    for (int i = tl.first + 1; i < br.first; ++i) {
        for (int j = tl.second + 1; j < br.second; ++j) {
            ans += (mp[i][j] == '@');
        }
    }

    cout << ans;

    return 0;
}
