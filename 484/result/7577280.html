class Layer{
public:
    int nodesIn;
    int nodesOut;
    vector<vector<double>> weights;
    vector<double> biases;
    bool P1;
 
    Layer(){}
    Layer(int _nodesIn, int _nodesOut, vector<vector<double>> we, vector<double> bi, bool isP1){
        nodesIn = _nodesIn;
        nodesOut = _nodesOut;
        weights = we;
        biases = bi;
        P1 = isP1;
    }
 
    vector<double> calculate_outputs(vector<double> inputs){
        vector<double> outputs(nodesOut);
        for(int outI = 0; outI < nodesOut; outI++){
            double weight = biases[outI];
            for(int inI = 0; inI < nodesIn; inI++) weight += inputs[inI] * weights[inI][outI];
            outputs[outI] = P1 ? activation_function_P1(weight) : activation_function_P2(weight);
        }
        return outputs;
    }
};
