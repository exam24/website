#include<bits/stdc++.h>
using namespace std;

int net[509][1009];
int sum[509][1009];

int sumTotal;
int n,m,k;
int ansTri=1e9;

int fixB(int a,int b){
	if(a%2==0)return 2*b;
	return 2*b-1;
}

int main(){
	cin>>n>>m>>k;
	m=m*2;
	for(int i=1;i<=k;i++){
		int a,b;
		char c;
		cin>>a>>b>>c;
		if(c=='H'){
			net[a][fixB(a,b)]=1;
		}
		else if(c=='K'){
			net[a][fixB(a,b)]=-10;
		}
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			sumTotal+=net[i][j];
			sum[i][j]=sum[i][j-1]+net[i][j];
		}
	}
	
	for(int i=1;i<=n;i++){
		for(int j2=1;j2<=m/2;j2++){
			int j=fixB(i,j2);
			int sz=1;
			int sumUpw=0,sumDnw=0;
			while(i+sz-1<=n&&j-sz>=0&&j+sz-1<=m){
				sumUpw=sumUpw+(sum[i+sz-1][j+sz-1]-sum[i+sz-1][j-sz]);
				ansTri=min(ansTri,sumUpw);
				sz++;
//				cout<<sumUpw<<endl;
			}
			sz=1;
			while(i-sz+1>=1&&j-sz>=0&&j+sz-1<=m){
				sumDnw=sumDnw+(sum[i-sz+1][j+sz-1]-sum[i-sz+1][j-sz]);
				ansTri=min(ansTri,sumDnw);
				sz++;
//				cout<<sumDnw<<endl;
			}
		}
	}
	
//	for(int i=1;i<=n;i++){
//		for(int j=1;j<=m;j++){
//			if(net[i][j]==-10)cout<<"X ";
//			else cout<<net[i][j]<<" ";
//		}
//		cout<<endl;
//	}

	cout<<sumTotal-ansTri<<endl;
	return 0;
}
