#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

typedef long long ll;

ll n, m, k;

vector<vector<ll>> g;
vector<ll> dist;
vector<ll> p;
vector<ll> sz;
vector<ll> ma;

bool cmp(ll a, ll b) {
    return dist[b] < dist[a];
}

ll get(ll x) {
    if (p[x] == x) {
        return x;
    } else {
        return p[x] = get(p[x]);
    }
}

void unite(ll a, ll b) {
    a = get(a);
    b = get(b);
    if (a == b) {
        return;
    }
    /*if (sz[a] > sz[b]) {
        ll tmp = a;
        a = b;
        b = tmp;
    } */
    p[a] = b;
}

int main() {
    ll n; cin >> n;

    vector<ll> vert(n);
    p.resize(n);
    sz.resize(n);
    for (ll i = 0; i < n; i++) {
        vert[i] = i;
        p[i] = i; sz[i] = 1;
    }

    g.resize(n);
    for (ll i = 0; i < n - 1; i++) {
        ll u, v; cin >> u >> v;
        g[u - 1].push_back(v - 1);
    }

    queue<ll> q;
    dist.resize(n);
    q.push(0); dist[0] = 0;
    while (q.size() > 0) {
        ll v = q.front();
        q.pop();
        for (ll u : g[v]) {
            dist[u] = dist[v] + 1;
            q.push(u);
        }
    }

    ma.resize(n);
    for (ll i = 0; i < n; i++) {
        cin >> ma[i];
    }

    sort(vert.begin(), vert.end(), cmp);

    //cout << vert.size() << ' ';
    for (ll v : vert) {
        ll sum = 0;
        for (ll u : g[v]) {
            unite(v, u);
            sum += ma[u];
        }
        ma[v] = max(ma[v], sum);
        //cout << v << ' ' << ma[v] << '\n';
    }
    cout << ma[0];
}
